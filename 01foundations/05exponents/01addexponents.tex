\begin{exercise}{Potensregneregler 1-1}

Bestem $a^3 \cdot a^5$.

\answerbox{a^8}

\hint

Benyt formlen
\[
x^n \cdot x^m = x^{n+m}
\]

\hint

Indsæt $x=a$, $n=3$ og $m=5$.
\[
a^3 \cdot a^5 = a^{3+5} = a^8
\]

\end{exercise}
