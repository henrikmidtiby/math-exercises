\begin{exercise}{Logaritme basis viden 1}

Bestem $\log_{3}(9)$.

\answerbox{2}

\hint
Hvor mange gange skal 3 ganges med sig selv for at give 9?

\hint
\[
3 \cdot 3 = 3^2 = 9
\]

\hint
Da 3 skal ganges med sig selv to gange er 
$\log_3(9) = 2$.

\end{exercise}
