\begin{exercise}{Skifte af logaritme baser 1}

Bestem $\log_{16}(8)$ ved at skifte basen for logaritmen til 2.

\answerbox{3/4}

\hint

Anvend formlen 
\[
\log_a(b) = \frac{\log_c(b)}{\log_c(a)}
\]

\hint

Indsæt $a=16$, $b=8$ og $c=2$.

\hint
\[
\log_{16}(8) = \frac{\log_2(8)}{\log_2(16)}
\]

\hint
Bestem $\log_2(8)$.

\hint
\[
\log_2(8) = 3
\]

\hint
Bestem $\log_2(16)$.

\hint
\[
\log_2(16) = 4
\]

\hint
\[
\log_{16}(8) = \frac{3}{4}
\]


\end{exercise}
