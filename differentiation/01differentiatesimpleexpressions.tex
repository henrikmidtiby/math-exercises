\begin{exercise}{Differentiering 1 - 01}

Bestem $\frac{d}{dx} x^2$.

\answerbox{2x}

\hint

Benyt formlen
\[
\frac{d}{dx} x^n = n \cdot x^{n - 1}
\]

\hint

Sæt $n = 2$
\[
\frac{d}{dx} x^2 = 2 \cdot x^{2 - 1} = 2 x
\]
\end{exercise}
